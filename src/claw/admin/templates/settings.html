Usage Examples

                
# Send a message
curl -X POST http://HOST:8080/api/webhook \
  -H "Content-Type: application/json" \
  -d '{"type":"message","payload":{"text":"Turn off the lights"},"source":"home-assistant"}'

# Send a notification
curl -X POST http://HOST:8080/api/webhook \
  -H "Content-Type: application/json" \
  -d '{"type":"notification","payload":{"title":"Alert","message":"Front door opened"}}'

# Create a reminder
curl -X POST http://HOST:8080/api/webhook \
  -H "Content-Type: application/json" \
  -d '{"type":"reminder","payload":{"message":"Check laundry","time":"in 30 min"}}'
